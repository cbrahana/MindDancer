CREATE TABLE NODE (
--Identification Sections
    "NIN"        INTEGER  PRIMARY KEY; 
--Utility Sections
    "PIN"        INTEGER  NOT NULL,
    "NAME"       TINYTEXT,
    "FILE"       VARCHAR(4096), --Max length for filenames in linux
    "NOTE"       MEDIUMTEXT,
--Position; is this really  necessary? Or do we want ad-hoc setup. NOTE 1.
    "X_POSITION" INT,
    "Y_POSITION" INT,
    "Z_POSITION" INT,

    "HYPERLINK"  MEDIUMTEXT,
    "IMAGE_LINK" MEDIUMTEXT,

    PRIMARY KEY (NIN)
);

--Creates Table to Describe links
CREATE TABLE LINK (
    "LINK_ONE"   INT  NOT NULL,
    "LINK_TYPE"  SMALLINT,
    "LINK_TWO"   INT  NOT NULL,
);
