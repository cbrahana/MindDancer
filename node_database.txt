CREATE TABLE NODE (
--Identification Sections
  "NIN"        INTEGER,
  "PIN"        INTEGER  NOT NULL,
  "NAME"       TINYTEXT,

  PRIMARY KEY (NIN)
);

--Creates Table to Describe links
CREATE TABLE LINK (
  "LINK_ONE"   INTEGER  NOT NULL;
  "LINK_TWO"   INTEGER  NOT NULL;
  "LINK_TYPE"  MEDIUMINTEGER  NOT NULL;
);
